## Overview

Build the protected `/admin` route on the web app. Admin (Firebase custom claim
`admin: true`) can view all videos, add new ones via a slide-over form, toggle
`isActive` per video, and send push notification triggers.

**Tech stack:** React 18 + Vite, TypeScript, Vitest (unit), Playwright (E2E)
**Test runner:** `vitest run` (unit) | `FIREBASE_EMULATOR_RUNNING=1 npx playwright test` (E2E)
**Lint/type check:** `npm run lint` | `npm run check:types`

---

## Known Test Gaps

- **Notification real-time status (FT-8):** The Cloud Function that transitions
  a notification from `pending` â†’ `sent` does not exist yet (Story 9 scope).
  FT-8 simulates the status change by directly writing to the Firestore emulator
  via REST and asserting the UI reflects the change via `onSnapshot`. This is
  the correct approach â€” it tests the real-time listener in isolation without
  requiring the Cloud Function.

- **`getIdTokenResult` forceRefresh in emulator:** Custom claims set via the
  emulator REST API before sign-in are embedded in the initial JWT, so
  `getIdTokenResult(user, true)` will return the correct claims on first call
  without a network round-trip. No additional test hook is needed.

const notificationConverter: FirestoreDataConverter<Notification> = {
  toFirestore(n: Notification) {
    const { notificationId: _id, ...rest } = n
    return rest
  },
  fromFirestore(snapshot, options): Notification {
    return { notificationId: snapshot.id, ...snapshot.data(options) } as Notification

```tsx
// AdminGuard.tsx
import { Navigate } from 'react-router-dom'
import { useState, useEffect } from 'react'
import { getIdTokenResult } from 'firebase/auth'
import { useAuthStore } from '../store/authStore'
import { LoadingSpinner } from './LoadingSpinner'

export function AdminGuard({ children }: { children: React.ReactNode }) {
  const { user, loading: authLoading } = useAuthStore()
  const [adminStatus, setAdminStatus] = useState<'loading' | 'admin' | 'denied'>('loading')

  useEffect(() => {
    if (authLoading) return
    if (!user) { setAdminStatus('denied'); return }
    getIdTokenResult(user, true)
      .then((result) => setAdminStatus(result.claims['admin'] === true ? 'admin' : 'denied'))
      .catch(() => setAdminStatus('denied'))
  }, [user, authLoading])

  if (authLoading || adminStatus === 'loading') return <LoadingSpinner />
  if (adminStatus === 'denied') return <Navigate to="/" replace />
  return <>{children}</>
}
```

}

```ts
// adminService.ts
import {
  getDocs, query, orderBy, addDoc, updateDoc, doc,
  onSnapshot, serverTimestamp, Unsubscribe
} from 'firebase/firestore'
import { db } from '@ckd/shared/firebase/config'
import { videosCollection } from '@ckd/shared/firebase/collections'
import { notificationsCollection } from '@ckd/shared/firebase/collections'
import type { Video, Category } from '@ckd/shared/types/video'
import type { Notification } from '@ckd/shared/types/notification'

export interface AddVideoInput {
  youtubeVideoId: string
  title: string
  category: Category
  thumbnailUrl: string
  durationSeconds: number
  order: number
}

export interface AddNotificationInput {
  title: string
  body: string
  youtubeVideoId: string | null
}

// Fetch all videos (active and inactive) ordered by publishedAt DESC
export async function fetchAllVideos(): Promise<Video[]> {
  const q = query(videosCollection(db), orderBy('publishedAt', 'desc'))
  const snap = await getDocs(q)
  return snap.docs.map((d) => d.data())
}

// Get max order value for a category (for auto-ordering new videos)
export async function getMaxOrderForCategory(category: Category): Promise<number> {
  const all = await fetchAllVideos()
  const inCategory = all.filter((v) => v.category === category)
  return inCategory.length === 0 ? 0 : Math.max(...inCategory.map((v) => v.order))
}

// Add a new video document
export async function addVideo(input: AddVideoInput): Promise<void> {
  await addDoc(videosCollection(db), {
    ...input,
    isActive: true,
    publishedAt: serverTimestamp(),
  } as unknown as Video)
}

// Toggle isActive for a video
export async function toggleIsActive(videoId: string, currentIsActive: boolean): Promise<void> {
  await updateDoc(doc(db, 'videos', videoId), { isActive: !currentIsActive })
}

// Write notification document to trigger Cloud Function
export async function addNotification(input: AddNotificationInput): Promise<string> {
  const ref = await addDoc(notificationsCollection(db), {
    ...input,
    createdAt: serverTimestamp(),
    sentAt: null,
    status: 'pending',
  } as unknown as Notification)
  return ref.id
}

// Real-time listener on a notification document
export function subscribeToNotification(
  notificationId: string,
  callback: (n: Notification | null) => void,
): Unsubscribe {
  const ref = doc(notificationsCollection(db), notificationId)
  return onSnapshot(ref, (snap) => {
    callback(snap.exists() ? snap.data() : null)
  })
}
```

**Test Requirements (against Firestore emulator):**
- `fetchAllVideos()` with 2 seeded docs â†’ returns both, newest first
- `addVideo(input)` â†’ new doc in Firestore with `isActive: true`
- `toggleIsActive(id, true)` â†’ doc has `isActive: false`
- `addNotification(input)` â†’ doc with `status: 'pending'`, `sentAt: null`
- `subscribeToNotification(id, cb)` â†’ callback fires with updated data when doc changes

**Implementation Notes:**
- `fetchAllVideos` has NO `where('isActive', '==', true)` â€” admin sees ALL videos
- `serverTimestamp()` used for `publishedAt` and `createdAt`
- `addVideo` cast to `unknown as Video` needed because `publishedAt` is `serverTimestamp()` not a real Timestamp at write time

### Task A-5 of 10: `useAdmin.ts` Hook

**Type:** Feature

**Files:**
- Test: `apps/web/src/features/admin/__tests__/useAdmin.test.ts` (new)
- Implementation: `apps/web/src/features/admin/hooks/useAdmin.ts` (new)

**What to Build:**

```ts
// useAdmin.ts
import { useState, useEffect, useCallback, useRef } from 'react'
import type { Video } from '@ckd/shared/types/video'
import type { Notification } from '@ckd/shared/types/notification'
import type { AddVideoInput, AddNotificationInput } from '../services/adminService'
import {
  fetchAllVideos, addVideo, toggleIsActive,
  addNotification, subscribeToNotification, getMaxOrderForCategory
} from '../services/adminService'

export interface UseAdminReturn {
  videos: Video[]
  loading: boolean
  error: string | null
  publishVideo: (input: Omit<AddVideoInput, 'order'>) => Promise<void>
  toggleVideoActive: (videoId: string, currentIsActive: boolean) => Promise<void>
  sendNotification: (input: AddNotificationInput) => Promise<void>
  latestNotification: Notification | null
  notificationLoading: boolean
}

export function useAdmin(): UseAdminReturn {
  const [videos, setVideos] = useState<Video[]>([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const [latestNotification, setLatestNotification] = useState<Notification | null>(null)
  const [notificationLoading, setNotificationLoading] = useState(false)
  const unsubRef = useRef<(() => void) | null>(null)

  const loadVideos = useCallback(async () => {
    setLoading(true)
    setError(null)
    try {
      const vids = await fetchAllVideos()
      setVideos(vids)
    } catch (e) {
      setError('Failed to load videos')
    } finally {
      setLoading(false)
    }
  }, [])

  useEffect(() => { void loadVideos() }, [loadVideos])

  // Clean up notification listener on unmount
  useEffect(() => () => { unsubRef.current?.() }, [])

  const publishVideo = useCallback(
    async (input: Omit<AddVideoInput, 'order'>) => {
      const order = await getMaxOrderForCategory(input.category)
      await addVideo({ ...input, order: order + 1 })
      await loadVideos()
    },
    [loadVideos],
  )

  const toggleVideoActive = useCallback(
    async (videoId: string, currentIsActive: boolean) => {
      await toggleIsActive(videoId, currentIsActive)
      await loadVideos()
    },
    [loadVideos],
  )

  const sendNotification = useCallback(async (input: AddNotificationInput) => {
    setNotificationLoading(true)
    unsubRef.current?.()
    const id = await addNotification(input)
    const unsub = subscribeToNotification(id, (n) => {
      setLatestNotification(n)
      if (n?.status !== 'pending') setNotificationLoading(false)
    })
    unsubRef.current = unsub
  }, [])

  return {
    videos, loading, error,
    publishVideo, toggleVideoActive, sendNotification,
    latestNotification, notificationLoading,
  }
}
```

**Test Requirements:**
- Initial render â†’ `loading: true` â†’ after fetch resolves â†’ `loading: false`, `videos` populated
- `publishVideo(input)` â†’ calls `addVideo` then re-fetches
- `toggleVideoActive(id, true)` â†’ calls `toggleIsActive(id, true)` then re-fetches
- `sendNotification(input)` â†’ calls `addNotification`, starts subscription
- Unsubscribes from notification listener on unmount

**Implementation Notes:**
- `loadVideos` wrapped in `useCallback` to avoid stale closure in `publishVideo`/`toggleVideoActive`
- `notificationLoading` set to false when status transitions away from `pending`
